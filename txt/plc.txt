	rdef.inc
			#define   R_LUBRICA_FREQ_CAMPLUS   6231 // Lubrica: freq da camplus, variabile, impostabile in base alle necessitŕ di lubrificazione del pezzo da lavorare
			#define   R_LUBRICA_DUTY_CYCLE     6232 // Lubrica: duty cycle
			#define   R_LUBRICA_PERIODO        6233 // Lubrica: periodo, durata di ogni ciclo di lubrificazione (tempo pompa ON + tempo pompa OFF)
			#define   R_LUBRICA_TEMPO_ON       6234 // Lubrica: tempo pompa ON

														// 0: (old) la freq. impostata in Camplus varia la durata del tempo in cui la pompa č OFF e la frequenza di lubrificazione (il tempo di pompa ON č fisso a 0,3s)
														// 1: (new) la freq. impostata in Camplus varia il duty cycle, la % di tempo in cui la pompa č ON. Mentre la frequenza di lubrificazione resta costante (f=R7018)

			#define   R_LUBRICA_FREQ_CONFMAC       7018 // Lubrica: (considerata se R7017=1) frequenza fissa dei cicli di lubrificazione da confmac (30 cicli/min  =>  T=2sec) occorre mantenere bassi i cicli di attivazioni al minuto per allungare la vita dell’impianto di lubrificazione

	lubrica.c
		R_LUBRICA_TIPO_GESTIONE determina la gestione attiva,
		eliminati  float TIME;  double APP_FREQ;
		Nuova gestione lubrificazione:
		- in CONFMAC si imposta la frequenza fissa di lubrificazione (R_LUBRICA_FREQ_CONFMAC) : cicli di lubrificazione al minuto, occorre lasciar impostato 30 cicli al minuto,
		- in base a quest'ultima frequenza il PLC calcola il periodo, la durata fissa di ogni ciclo (ON+OFF) di lubrificazione (R_LUBRICA_PERIODO)
		  se R_LUBRICA_FREQ_CONFMAC=30  =>  la durata fissa di ogni ciclo (ON+OFF) di lubrificazione sarŕ di 2sec.
		- in Camplus/Hdrill si continua ad impostare una teorica freq.di lubrificazione (R_FREQ_LUBRICA) 
		- in base a quest'ultima teorica freq.di lubrificazione il PLC determina:
		  * il duty-cycle (R_LUBRICA_DUTY_CYCLE) la % di tempo di ogni ciclo di lubrificaz. in cui la pompa lubrica deve esser ON,
		  * il tempo in sec. di ogni ciclo di lubrificaz. in cui la pompa č ON  (R_LUBRICA_TEMPO_ON) 
		  * il tempo in sec. di ogni ciclo di lubrificaz. in cui la pompa č OFF (R_LUBRICA_TEMPO_OF) 
		  * reimposta i timer (T_LUBRICA_ON T_LUBRICA_OFF) di gestione lubrica in base a questi 2 tempi
		- il pilotaggio dei timer di lubrificazione č ora sincronizzato anche col pulsante HOLD.
		- se (in Camplus/Hdrill) si imposta freq. di lubrificazione = 0   la pompa rimane sempre OFF
		- se (in Camplus/Hdrill) si imposta freq. di lubrificazione = 1   la pompa rimane ON nel 20% di ogni ciclo di lubrificazione
		- se (in Camplus/Hdrill) si imposta freq. di lubrificazione = da 2 a 118 la pompa rimane ON per una % di tempo, di ogni ciclo di lubrificazione, proporzionale alla freq.
		- se (in Camplus/Hdrill) si imposta freq. di lubrificazione = 119 la pompa rimane ON nel 80% di ogni ciclo di lubrificazione
		- se (in Camplus/Hdrill) si imposta freq. di lubrificazione = 120 la pompa rimane sempre ON 
		
		Dutycycle = 20 + f*0.5   (formula valida in tutti i casi tranne i 2 casi particolari iniziali e finali)
		
		f    =>   Dutycycle
		0    =>   0
		1    =>   20
		10   =>   25
		20   =>   30
		30   =>   35
		40   =>   40
		50   =>   45
		60   =>   50
		70   =>   55
		80   =>   60
		90   =>   65
		100  =>   70
		110  =>   75
		119  =>   80
		120  =>   100
